! Retira a acl aplicada
interface Vlan%NUMERO
 no ip access-group %ACLout out
 no ip access-group %ACLin in

! Limpa acl out
no ip access-list extended %ACLout

! Cria acl out
ip access-list extended %ACLout

! Acessos ICMP

! Acesso icmp  FE
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  echo
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  echo-reply
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  packet-too-big
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  unreachable
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  traceroute
permit icmp 172.16.0.0 0.15.255.255  %REDE %WMASC  time-exceeded

! Bloqueio anti-spoof
deny ip  %REDE %WMASC  any

! Acessos TCP

!B05-DC-RJ e B06-DC-RJ
permit tcp 172.31.9.0 0.0.0.255 %REDE %WMASC eq 80
permit tcp 172.31.9.0 0.0.0.255 %REDE %WMASC eq 443
permit tcp 172.31.9.0 0.0.0.255 %REDE %WMASC range 8000 8100

permit tcp 172.31.10.0 0.0.0.255 %REDE %WMASC eq 80
permit tcp 172.31.10.0 0.0.0.255 %REDE %WMASC eq 443
permit tcp 172.31.10.0 0.0.0.255 %REDE %WMASC range 8000 8100



!Caches Staging
permit tcp 192.168.69.0 0.0.0.255   %REDE %WMASC  eq 80
permit tcp 192.168.69.0 0.0.0.255   %REDE %WMASC  eq 443
permit tcp 192.168.69.0 0.0.0.255   %REDE %WMASC  range 8000 8100


! Monitoracao
permit tcp 172.16.55.0 0.0.0.255   %REDE %WMASC  eq 80
permit tcp 172.16.55.0 0.0.0.255   %REDE %WMASC  eq 443
permit tcp 172.16.55.0 0.0.0.255   %REDE %WMASC  range 8000 8100


! NAT CORP-FE
permit tcp 172.31.255.128 0.0.0.127   %REDE %WMASC 

! Bloqueia outras Redes Privadas
deny tcp 10.0.0.0 0.255.255.255   %REDE %WMASC  
deny tcp 172.16.0.0 0.15.255.255  %REDE %WMASC  
deny tcp 192.168.0.0 0.0.255.255  %REDE %WMASC  

! Permite mundo
permit tcp any   %REDE %WMASC  eq 80
permit tcp any   %REDE %WMASC  eq 443

! Bloqueia ALL
deny ip any any

! Limpa acl
no ip access-list extended %ACLin

! Cria acl in
ip access-list extended %ACLin

! Acesso hsrp
permit udp host  %ESPECIAL1 eq 1985 host 224.0.0.2 eq 1985
permit udp host  %ESPECIAL2 eq 1985 host 224.0.0.2 eq 1985

! Acessos ICMP

! Acesso icmp NAT FE
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 echo
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 echo-reply
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 packet-too-big
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 unreachable
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 traceroute
permit icmp  %REDE %WMASC  172.16.0.0 0.15.255.255 time-exceeded

! Bloqueio anti-spoof
deny ip any %REDE %WMASC 

! Retorno de acessos
permit tcp  %REDE %WMASC  any established

! Bloqueia ALL
deny ip any any


! Reaplica as ACL's
interface vlan%NUMERO
  ip access-group %ACLout out
  ip access-group %ACLin in
  no shutdown
!
end
